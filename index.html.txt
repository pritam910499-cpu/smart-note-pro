<script>
function downloadPDF(){
    const text = document.getElementById('note').value;
    const blob = new Blob([text], {type:'text/plain'});
    const link = document.createElement('a');
    link.href = URL.createObjectURL(blob);
    link.download = "note.txt";
    link.click();
}
</script>
